CATIA Automation

As model complexity increases, manual workflows in CATIA become time-consuming and error-prone. Automation ensures consistency, repeatability, and control across large feature trees. It reduces engineering time while minimizing human error in design and downstream processes. Ultimately, it enables faster iterations and more scalable, production-ready workflows.

Field
Mechanical Engineering
Tools
C#, CATIA, Visual-Studio
Focus
CAD/CAM Automation

Project Overview

When working with complex models containing a large number of features, maintaining a consistent and systematic naming convention becomes critical—especially when transitioning into CAM processes. I developed a script to automate entity naming within CATIA, ensuring structured and reliable model organization. This solution saves significant engineering time and enables efficient model updates without the need for repetitive manual renaming.

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;


using INFITF;
using MECMOD;
using PARTITF;
using ProductStructureTypeLib;
using NavigatorTypeLib;
using HybridShapeTypeLib; 

namespace CatiaNameChanging
{
    public partial class Form1 : Form
    {
        public INFITF.Application catia;
        public int o_from;
        public int o_to;
        public int n_from;
        public String oldname;
        public String gsa;
        public String newname;

        public int a1 = 51;
        public int del = 30;
        public int a12 ;
        public int a2 = 135;
        public int a22 = 162;
        public int from = 1251;

        public int g1 = 41;
        public Form1()
        {
            InitializeComponent();

            if (DateTime.UtcNow.Year != 2025)
                button1.Enabled = false;  
        }

        private void button1_Click(object sender, EventArgs e)
        {
            //autorun();   // specific autorun
            //autorun2(); // special run mode
            o_from = int.Parse(textBox3.Text);
            o_to = int.Parse(textBox4.Text);
            n_from = int.Parse(textBox7.Text);

           
            gsa = textBox1.Text;
            oldname = textBox2.Text;
            newname = textBox5.Text;

 
            catia = (INFITF.Application)System.Runtime.InteropServices.Marshal.GetActiveObject("Catia.Application");
            PartDocument pd = (PartDocument)catia.ActiveDocument;
            Part pt = pd.Part;

 
            HybridBodies hybridBodies1 = pt.HybridBodies;
            HybridBody hybridBody1 = hybridBodies1.Item("Entity_Type2_NUM_s");
            HybridBodies hybridBodies2 = hybridBody1.HybridBodies;
            HybridBody hybridBody2 = hybridBodies2.Item(gsa);
            HybridBodies hybridBodies3 = hybridBody2.HybridBodies;

            int z = n_from;
            for (int i = o_from; i  o_to+1; i++)
            {
                HybridShape hybridBody3 = hybridBody2.HybridShapes.Item(String.Concat(oldname, i.ToString()));
                hybridBody3.set_Name(String.Concat(newname, z.ToString()));
                z++;
            }
        }


        public void autorun()
        {
            if(g1%2!=0)

            {
                textBox1.Text = String.Concat("Entity_NUM_.", g1.ToString());
                textBox5.Text = textBox2.Text = "Entity_Type2_NUM__";
                textBox3.Text = a1.ToString();
                textBox4.Text = a12.ToString();
                textBox7.Text = from.ToString();
                g1++;
                from = from+del;
            }

            else

            {
                textBox1.Text = String.Concat("Entity_NUM_.", g1.ToString());
                textBox5.Text = textBox2.Text = "Entity_Type2_NUM__";
                textBox3.Text = a2.ToString();
                textBox4.Text = a22.ToString();
                textBox7.Text = from.ToString();
                g1++;
                from = from + del;

            }
        }

        public void autorun2()
        {
                textBox1.Text = String.Concat("Entity_NUM_.", g1.ToString());
                textBox5.Text = textBox2.Text = "Entity_Type2_NUM__";
                textBox3.Text = a1.ToString();
                textBox4.Text = (a1+del-1).ToString();
                textBox7.Text = from.ToString();
            
                g1++;
                a1 = a1 + del;
                from = from + del;
        }
     }
}
